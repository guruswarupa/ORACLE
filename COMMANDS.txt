Data Definition Language (DDL)

create a new table: CREATE TABLE CUSTOMERS(id int,name VARCHAR(30),age int,address VARCHAR(80),salary decimal (10,2));

check table structure: DESC CUSTOMERS;

alter table:
    modify:  ALTER TABLE CUSTOMERS MODIFY salary decimal(18,2) not null;
    add:     ALTER TABLE CUSTOMERS ADD mobile int not null;

drop/remove table:  DROP TABLE customers;

primary key: CREATE TABLE CUSTOMERS(id int not null, name varchar2(20) not null, age int not null, address varchar2(25),salary decimal(18,2),primary key(id));

add constraint: ALTER TABLE CUSTOMERS ADD CONSTRAINT customCheck CHECK(age >= 18);

drop constraint: ALTER TABLE CUSTOMERS DROP CONSTRAINT customCheck;

Data Manipulation Language (DML)

select data from table: SELECT * FROM CUSTOMERS;
insert data into table: INSERT INTO CUSTOMERS(id,name,age,address,salary) values (1,'Guru Swarupa',20,'#13 srivasa, marathahalli, bengaluru 560037',120000);
update data from table: UPDATE CUSTOMERS set name = 'Tarun' WHERE ID=1; 
delete data from table: DELETE FROM CUSTOMERS WHERE id=1;
