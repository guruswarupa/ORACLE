1, docker pull gvenzl/oracle-xe

2, docker run -d \
  --name oracle-xe \
  -p 1521:1521 \
  -p 8080:8080 \
  -e ORACLE_PASSWORD=YourStrongPassword \
  gvenzl/oracle-xe

3, docker exec -it oracle-xe sqlplus system/YourStrongPassword@//localhost:1521/XEPDB1

4, CREATE USER myuser IDENTIFIED BY mypass;

5, GRANT CONNECT, RESOURCE TO myuser;

6, GRANT CREATE SESSION TO myuser;

7, exit and you can login to user myuser.

8, docker exec -it oracle-xe sqlplus myuser/mypass@//localhost:1521/XEPDB1
